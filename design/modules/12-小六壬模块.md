# 小六壬模块

> **所属文档**：方案设计 · 子模块  
> **版本**：v1.0

## 1. 模块定位

- **方法**：小六壬（掐指一算） — 6 宫循环推演，月/日/时三步得位
- **可行性**：25/25 分 🟢 强推荐（见可行性分析 §3.1 方法 2）
- **体验定位**："今日运势"快测入口，1 分钟出结果

## 2. 数据结构

```typescript
type PalaceName = '大安' | '留连' | '速喜' | '赤口' | '小吉' | '空亡'
type JiXiong = 'ji' | 'xiong' | 'neutral'
type ElementCN = '木' | '火' | '土' | '金' | '水'
type Aspect = 'overall' | 'career' | 'love' | 'wealth' | 'health'

interface PalaceReading {
  name: PalaceName
  ji_xiong: JiXiong
  element: ElementCN
  symbol: string                            // emoji 或图标
  readings: Record<Aspect, string>          // 多维度解读
  suitable: string[]                        // 宜
  avoid: string[]                           // 忌
}

interface LiurenResult {
  palace: PalaceReading                     // 最终宫位
  steps: {
    month: number                            // 农历月数
    day: number                              // 农历日数
    hour: number                             // 时辰序号
  }
  path: PalaceName[]                        // 三步经过的宫位
}
```

## 3. 算法

```text
1. BirthInfo (或"今日" + 当前时辰) → 农历月、日、时辰序号（1-12）
2. 起点 = '大安'（索引 0）
3. 第 1 步：从'大安'数农历月份步 → P1
4. 第 2 步：从 P1 数农历日数步 → P2
5. 第 3 步：从 P2 数时辰序号步 → P3
6. P3 即为最终宫位
```

### 3.1 关键代码

```typescript
const PALACES_ORDER: PalaceName[] = [
  '大安', '留连', '速喜', '赤口', '小吉', '空亡',
]

export function calculateLiuren(month: number, day: number, hour: number): LiurenResult {
  let pos = 0
  const path: PalaceName[] = []

  pos = (pos + month - 1) % 6
  path.push(PALACES_ORDER[pos])

  pos = (pos + day - 1) % 6
  path.push(PALACES_ORDER[pos])

  pos = (pos + hour - 1) % 6
  path.push(PALACES_ORDER[pos])

  const finalName = PALACES_ORDER[pos]
  const palace = PALACES[finalName]

  return {
    palace,
    steps: { month, day, hour },
    path,
  }
}
```
