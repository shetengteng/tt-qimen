# 黄历择日模块

> **所属文档**：方案设计 · 子模块  
> **版本**：v1.0

## 1. 模块定位

- **方法**：传统黄历宜忌 — 根据每日吉神凶神、十二建星，推荐黄道吉日
- **可行性**：21/25 分 🟡 可做（见可行性分析 §3.2 方法 16）
- **体验定位**：工具型 — 用户查"今日宜忌" / 找黄道吉日做某事

## 2. 数据结构

```typescript
interface HuangliDay {
  date: string            // 'YYYY-MM-DD'
  lunarDate: string       // '乙巳年三月初一'
  ganZhi: {
    year: string
    month: string
    day: string
  }
  zodiac: string          // 生肖

  // 十二建星（每日一个）：建、除、满、平、定、执、破、危、成、收、开、闭
  twelveStar: string

  // 宜 / 忌
  suitable: string[]      // ['嫁娶', '出行', '祭祀']
  avoid: string[]         // ['动土', '开市']

  // 吉神凶神
  luckyGods: string[]     // ['天德', '月德']
  evilGods: string[]

  // 冲煞
  conflict: string        // '冲猪(丁亥)煞东'

  // 黄黑道
  isHuangDao: boolean     // 黄道吉日
  constellation: string   // 西方星座
}

// 事由 → 宜忌关键词映射
interface MatterKeywords {
  matter: '嫁娶' | '搬家' | '开市' | '出行' | '祭祀' | '动土' | '安床' | '求财' | '破土'
  keywords: string[]      // 在 suitable 字段中匹配的关键词集
  description: string     // 事由说明
}
```

## 3. 算法

### 3.1 单日查询（`tyme4ts` 原生 API）

```typescript
import { SolarDay } from 'tyme4ts'

export function getHuangliDay(year: number, month: number, day: number): HuangliDay {
  const solar = SolarDay.fromYmd(year, month, day)
  const lunar = solar.getLunarDay()

  return {
    date: `${year}-${pad(month)}-${pad(day)}`,
    lunarDate: `${lunar.getLunarMonth().getLunarYear().getSixtyCycle().getName()}年${lunar.getLunarMonth().getName()}${lunar.getName()}`,
    ganZhi: {
      year:  lunar.getLunarMonth().getLunarYear().getSixtyCycle().getName(),
      month: lunar.getLunarMonth().getSixtyCycle().getName(),
      day:   lunar.getSixtyCycle().getName(),
    },
    zodiac: lunar.getLunarMonth().getLunarYear().getChineseZodiac().getName(),
    twelveStar: lunar.getDayOfficer().getName(),
    suitable: lunar.getRecommends().map(r => r.getName()),
    avoid: lunar.getAvoids().map(a => a.getName()),
    luckyGods: lunar.getJiShen().map(g => g.getName()),
    evilGods: lunar.getXiongShen().map(g => g.getName()),
    conflict: lunar.getConflict(),
    isHuangDao: isHuangDaoByStar(lunar.getDayOfficer()),  // 自定义判断
    constellation: getConstellation(month, day),
  }
}

function isHuangDaoByStar(dayOfficer: string): boolean {
  // 青龙、明堂、金匮、天德、玉堂、司命 = 黄道（吉）
  return ['青龙', '明堂', '金匮', '天德', '玉堂', '司命'].includes(dayOfficer)
}
```

### 3.2 事由筛选

```typescript
export function findGoodDaysByMatter(
  year: number,
  month: number,
  matter: MatterKeywords['matter']
): HuangliDay[] {
  const mapping = MATTER_KEYWORDS[matter]
  const days: HuangliDay[] = []

  for (let d = 1; d <= daysInMonth(year, month); d++) {
    const hd = getHuangliDay(year, month, d)
    if (mapping.keywords.some(k => hd.suitable.includes(k))) {
      days.push(hd)
    }
  }
  return days
}
```
